        // ===== 수율 계산 기능 =====
        private void CalculateYieldButton_Click(object sender, EventArgs e)
        {
            try
            {
                string selectedDisplay = columnSelectionComboBox.SelectedItem.ToString();
                string selectedColumn = columnDisplayNames.FirstOrDefault(x => x.Value == selectedDisplay).Key;
                
                decimal upperLimit = upperLimitUpDown.Value;
                decimal lowerLimit = lowerLimitUpDown.Value;

                if (upperLimit <= lowerLimit)
                {
                    MessageBox.Show("상한치는 하한치보다 커야 합니다!", "입력 오류", MessageBoxButtons.OK, MessageBoxIcon.Warning);
                    return;
                }

                // DB에서 데이터 조회
                int totalCount = 0;
                int okCount = 0;
                int ngCount = 0;

                using (MySqlConnection connection = new MySqlConnection(connectionString))
                {
                    connection.Open();

                    string query = $"SELECT {selectedColumn} FROM vertical_measurements WHERE {selectedColumn} IS NOT NULL";
                    
                    using (MySqlCommand cmd = new MySqlCommand(query, connection))
                    {
                        using (MySqlDataReader reader = cmd.ExecuteReader())
                        {
                            while (reader.Read())
                            {
                                totalCount++;
                                decimal value = reader.GetDecimal(0);
                                
                                if (value >= lowerLimit && value <= upperLimit)
                                {
                                    okCount++;
                                }
                                else
                                {
                                    ngCount++;
                                }
                            }
                        }
                    }
                }

                if (totalCount == 0)
                {
                    MessageBox.Show("분석할 데이터가 없습니다. 먼저 CSV를 임포트하세요.", "데이터 없음", MessageBoxButtons.OK, MessageBoxIcon.Information);
                    return;
                }

                decimal yieldRate = (decimal)okCount / totalCount * 100;

                // DB에 수율 결과 저장
                SaveYieldAnalysisToDatabase(selectedColumn, upperLimit, lowerLimit, totalCount, okCount, ngCount, yieldRate);

                // 결과 출력
                string result = $"\r\n========== 수율 분석 결과 ==========\r\n" +
                               $"항목: {selectedDisplay}\r\n" +
                               $"상한치: {upperLimit:F3} / 하한치: {lowerLimit:F3}\r\n" +
                               $"총 개수: {totalCount}개\r\n" +
                               $"OK: {okCount}개\r\n" +
                               $"NG: {ngCount}개\r\n" +
                               $"수율: {yieldRate:F2}%\r\n" +
                               $"분석 시간: {DateTime.Now:yyyy-MM-dd HH:mm:ss}\r\n" +
                               $"====================================";

                resultTextBox.AppendText(result);

                Console.WriteLine("=========================================");
                Console.WriteLine($"[YIELD ANALYSIS] {selectedDisplay}");
                Console.WriteLine($"Upper: {upperLimit:F3} / Lower: {lowerLimit:F3}");
                Console.WriteLine($"Total: {totalCount} | OK: {okCount} | NG: {ngCount}");
                Console.WriteLine($"Yield: {yieldRate:F2}%");
                Console.WriteLine("=========================================");

                MessageBox.Show($"수율 분석 완료!\n\n총: {totalCount}개\nOK: {okCount}개\nNG: {ngCount}개\n수율: {yieldRate:F2}%", 
                              "분석 완료", 
                              MessageBoxButtons.OK, 
                              MessageBoxIcon.Information);
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[ERROR] Yield calculation failed: {ex.Message}");
                MessageBox.Show($"수율 계산 오류!\n\n{ex.Message}", "오류", MessageBoxButtons.OK, MessageBoxIcon.Error);
            }
        }

        private void SaveYieldAnalysisToDatabase(string columnName, decimal upperLimit, decimal lowerLimit, 
                                                 int totalCount, int okCount, int ngCount, decimal yieldRate)
        {
            try
            {
                using (MySqlConnection connection = new MySqlConnection(connectionString))
                {
                    connection.Open();

                    string query = @"
                        INSERT INTO yield_analysis (
                            column_name, upper_limit, lower_limit, total_count, 
                            ok_count, ng_count, yield_rate, created_at
                        ) VALUES (
                            @column_name, @upper_limit, @lower_limit, @total_count, 
                            @ok_count, @ng_count, @yield_rate, @created_at
                        )
                    ";

                    using (MySqlCommand cmd = new MySqlCommand(query, connection))
                    {
                        cmd.Parameters.AddWithValue("@column_name", columnName);
                        cmd.Parameters.AddWithValue("@upper_limit", upperLimit);
                        cmd.Parameters.AddWithValue("@lower_limit", lowerLimit);
                        cmd.Parameters.AddWithValue("@total_count", totalCount);
                        cmd.Parameters.AddWithValue("@ok_count", okCount);
                        cmd.Parameters.AddWithValue("@ng_count", ngCount);
                        cmd.Parameters.AddWithValue("@yield_rate", yieldRate);
                        cmd.Parameters.AddWithValue("@created_at", DateTime.Now);
                        
                        cmd.ExecuteNonQuery();
                        
                        Console.WriteLine("[DB SAVE] Yield analysis saved successfully");
                    }
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[DB ERROR] Yield analysis save failed: {ex.Message}");
            }
        }

        // ===== CSV Import 기능 =====
        private void ImportCsvButton_Click(object sender, EventArgs e)
        {
            if (isImporting)
            {
                cancellationTokenSource?.Cancel();
                return;
            }

            OpenFileDialog openFileDialog = new OpenFileDialog
            {
                Filter = "CSV Files (*.csv)|*.csv|All Files (*.*)|*.*",
                Title = "Select CSV File"
            };

            if (openFileDialog.ShowDialog() == DialogResult.OK)
            {
                string filePath = openFileDialog.FileName;
                ImportCsvFile(filePath);
            }
        }

        private async void ImportCsvFile(string filePath)
        {
            try
            {
                List<MeasurementRecord> records = ReadCsvAndConvertToRecords(filePath);
                
                if (records.Count == 0)
                {
                    MessageBox.Show("CSV 파일에서 유효한 데이터를 찾을 수 없습니다.", 
                                  "오류", 
                                  MessageBoxButtons.OK, 
                                  MessageBoxIcon.Warning);
                    return;
                }

                isImporting = true;
                importCsvButton.Text = "취소";
                importCsvButton.BackColor = Color.FromArgb(255, 160, 122);
                progressBar.Visible = true;
                progressBar.Maximum = records.Count;
                progressBar.Value = 0;
                statusLabel.Visible = true;

                string logMessage = $"[{DateTime.Now:HH:mm:ss}] 총 {records.Count}개 행 로드 완료 (각 행당 18개 칼럼)";
                resultTextBox.AppendText($"\r\n{logMessage}");
                Console.WriteLine($"[CSV] Total {records.Count} rows loaded (18 columns each)");

                cancellationTokenSource = new CancellationTokenSource();
                var token = cancellationTokenSource.Token;

                int delayMs = 600;
                int savedCount = 0;
                int failedCount = 0;

                foreach (var record in records)
                {
                    if (token.IsCancellationRequested)
                    {
                        resultTextBox.AppendText($"\r\n[{DateTime.Now:HH:mm:ss}] 취소됨 ({savedCount}개 저장)");
                        break;
                    }

                    bool success = SaveMeasurementToDatabase(record);
                    
                    if (success)
                        savedCount++;
                    else
                        failedCount++;

                    progressBar.Value = savedCount + failedCount;
                    statusLabel.Text = $"진행: {savedCount + failedCount}/{records.Count} (성공: {savedCount}, 실패: {failedCount})";

                    await Task.Delay(delayMs, token);
                }

                string completeMessage = $"[{DateTime.Now:HH:mm:ss}] 완료 - 성공: {savedCount}, 실패: {failedCount}";
                resultTextBox.AppendText($"\r\n{completeMessage}");
                Console.WriteLine("=========================================");
                Console.WriteLine($"[IMPORT COMPLETE]");
                Console.WriteLine($"Total: {records.Count} rows | Saved: {savedCount} | Failed: {failedCount}");
                Console.WriteLine("=========================================");

                MessageBox.Show($"임포트 완료!\n\n총: {records.Count}개 행\n성공: {savedCount}개\n실패: {failedCount}개", 
                              "완료", 
                              MessageBoxButtons.OK, 
                              MessageBoxIcon.Information);
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[ERROR] {ex.Message}");
                Console.WriteLine($"[ERROR] Stack: {ex.StackTrace}");
                MessageBox.Show($"오류 발생!\n\n{ex.Message}", 
                              "오류", 
                              MessageBoxButtons.OK, 
                              MessageBoxIcon.Error);
            }
            finally
            {
                isImporting = false;
                importCsvButton.Text = "CSV 가져오기 (18개 칼럼)";
                importCsvButton.BackColor = Color.FromArgb(144, 238, 144);
                progressBar.Visible = false;
                statusLabel.Visible = false;
            }
        }

                    string query = @"

                            vertical_right_length_2, vertical_left_roundness_1, vertical_left_roundness_2,
                            vertical_right_roundness_1, vertical_right_roundness_2, vertical_left_angle_1, 
                            vertical_left_angle_2, vertical_right_angle_1, vertical_right_angle_2, created_at
                        ) VALUES (

        private List<MeasurementRecord> ReadCsvAndConvertToRecords(string filePath)
        {
            List<MeasurementRecord> records = new List<MeasurementRecord>();
            
            try
            {
                string csvContent = File.ReadAllText(filePath, Encoding.UTF8);
                List<string[]> csvData = ParseCsv(csvContent);
                
                Console.WriteLine($"[CSV] Total rows parsed: {csvData.Count}");
                
                if (csvData.Count == 0)
                    return records;

                string[] headers = csvData[0];
                Dictionary<string, int> columnMap = new Dictionary<string, int>();

                Console.WriteLine("[CSV] Column mapping for VERTICAL fields:");
                for (int i = 0; i < headers.Length; i++)
                {
                    string header = headers[i].Trim().ToUpper();
                    
                    if (!columnMap.ContainsKey(header))
                    {
                        columnMap[header] = i;
                        if (header.StartsWith("VERTICAL_"))
                        {
                            Console.WriteLine($"  Column[{i}] = {header}");
                        }
                    }
                }

                for (int rowIndex = 1; rowIndex < csvData.Count; rowIndex++)
                {
                    string[] columns = csvData[rowIndex];
                    
                    if (columns.Length == 0)
                        continue;

                    MeasurementRecord record = new MeasurementRecord
                    {
                        RowIndex = rowIndex,
                        VerticalNum = GetDecimalValue(columns, columnMap, "VERTICAL_NUM"),
                        VerticalUpperDiameter1 = GetDecimalValue(columns, columnMap, "VERTICAL_UPPER_DIAMETER_1"),
                        VerticalUpperDiameter2 = GetDecimalValue(columns, columnMap, "VERTICAL_UPPER_DIAMETER_2"),
                        VerticalUpperDiameterAvg = GetDecimalValue(columns, columnMap, "VERTICAL_UPPER_DIAMETER_AVG"),
                        VerticalLowerDiameter1 = GetDecimalValue(columns, columnMap, "VERTICAL_LOWER_DIAMETER_1"),
                        VerticalLowerDiameter2 = GetDecimalValue(columns, columnMap, "VERTICAL_LOWER_DIAMETER_2"),
                        VerticalLeftLength1 = GetDecimalValue(columns, columnMap, "VERTICAL_LEFT_LENGTH_1"),
                        VerticalLeftLength2 = GetDecimalValue(columns, columnMap, "VERTICAL_LEFT_LENGTH_2"),
                        VerticalRightLength1 = GetDecimalValue(columns, columnMap, "VERTICAL_RIGHT_LENGTH_1"),
                        VerticalRightLength2 = GetDecimalValue(columns, columnMap, "VERTICAL_RIGHT_LENGTH_2"),
                        VerticalLeftRoundness1 = GetDecimalValue(columns, columnMap, "VERTICAL_LEFT_ROUNDNESS_1"),
                        VerticalLeftRoundness2 = GetDecimalValue(columns, columnMap, "VERTICAL_LEFT_ROUNDNESS_2"),
                        VerticalRightRoundness1 = GetDecimalValue(columns, columnMap, "VERTICAL_RIGHT_ROUNDNESS_1"),
                        VerticalRightRoundness2 = GetDecimalValue(columns, columnMap, "VERTICAL_RIGHT_ROUNDNESS_2"),
                        VerticalLeftAngle1 = GetDecimalValue(columns, columnMap, "VERTICAL_LEFT_ANGLE_1"),
                        VerticalLeftAngle2 = GetDecimalValue(columns, columnMap, "VERTICAL_LEFT_ANGLE_2"),
                        VerticalRightAngle1 = GetDecimalValue(columns, columnMap, "VERTICAL_RIGHT_ANGLE_1"),
                        VerticalRightAngle2 = GetDecimalValue(columns, columnMap, "VERTICAL_RIGHT_ANGLE_2")
                    };

                    records.Add(record);
                }

                Console.WriteLine($"[CSV] {records.Count} records created from {csvData.Count - 1} data rows");
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[CSV ERROR] {ex.Message}");
                Console.WriteLine($"[CSV ERROR] Stack: {ex.StackTrace}");
            }

            return records;
        }

        private List<string[]> ParseCsv(string csvContent)
        {
            List<string[]> rows = new List<string[]>();
            List<string> currentRow = new List<string>();
            StringBuilder currentField = new StringBuilder();
            bool inQuotes = false;
            
            for (int i = 0; i < csvContent.Length; i++)
            {
                char c = csvContent[i];
                char nextChar = (i + 1 < csvContent.Length) ? csvContent[i + 1] : '\0';
                
                if (inQuotes)
                {
                    if (c == '"')
                    {
                        if (nextChar == '"')
                        {
                            currentField.Append('"');
                            i++;
                        }
                        else
                        {
                            inQuotes = false;
                        }
                    }
                    else
                    {
                        currentField.Append(c);
                    }
                }
                else
                {
                    if (c == '"')
                    {
                        inQuotes = true;
                    }
                    else if (c == ',')
                    {
                        currentRow.Add(currentField.ToString());
                        currentField.Clear();
                    }
                    else if (c == '\r' && nextChar == '\n')
                    {
                        currentRow.Add(currentField.ToString());
                        currentField.Clear();
                        rows.Add(currentRow.ToArray());
                        currentRow.Clear();
                        i++;
                    }
                    else if (c == '\n')
                    {
                        currentRow.Add(currentField.ToString());
                        currentField.Clear();
                        rows.Add(currentRow.ToArray());
                        currentRow.Clear();
                    }
                    else
                    {
                        currentField.Append(c);
                    }
                }
            }
            
            if (currentField.Length > 0 || currentRow.Count > 0)
            {
                currentRow.Add(currentField.ToString());
                rows.Add(currentRow.ToArray());
            }
            
            return rows;
        }

        private decimal? GetDecimalValue(string[] columns, Dictionary<string, int> columnMap, string columnName)
        {
            try
            {
                if (columnMap.ContainsKey(columnName))
                {
                    int index = columnMap[columnName];
                    if (index < columns.Length)
                    {
                        string value = columns[index].Trim();
                        
                        if (string.IsNullOrWhiteSpace(value) || 
                            value == "\\N" || 
                            value == "\\\\N" ||
                            value.Equals("NULL", StringComparison.OrdinalIgnoreCase))
                        {
                            return null;
                        }
                        
                        if (decimal.TryParse(value, out decimal result))
                        {
                            return result;
                        }
                        else
                        {
                            Console.WriteLine($"[WARNING] Cannot parse '{columnName}' value: '{value}'");
                            return null;
                        }
                    }
                }
                else
                {
                    Console.WriteLine($"[WARNING] Column '{columnName}' not found in CSV");
                }
            }
            catch (Exception ex)
            {
                Console.WriteLine($"[ERROR] GetDecimalValue for '{columnName}': {ex.Message}");
            }
            
            return null;
        }
